import DocCardList from '@theme/DocCardList';
import { useCurrentSidebarSiblings } from '@docusaurus/plugin-content-docs/client';

export function ReferenceCards() {
  const items = useCurrentSidebarSiblings().filter(
    item =>
      !(
        item.type === 'link' &&
        item.href.replace(/\/$/, '') === '/docs/reference'
      )
  );
  return <DocCardList items={items} />;
}

## What is reference documentation? {/* #what-is-reference-documentation */}

Reference documentation is **information-oriented** material
that describes the machinery of the product.
It's organized for easy lookup when you need specific technical details.

## Reference characteristics {/* #reference-characteristics */}

- **Information-focused**: Comprehensive technical details
- **Well-organized**: Easy to scan and find information
- **Accurate**: Precise and up-to-date specifications
- **Complete**: Covers all aspects of the topic
